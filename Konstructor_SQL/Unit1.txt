procedure TKonstryktor.Edit_BDClick(Sender: TObject);
begin

ADOTable2.FieldByName('Вариант 5').AsString:=V_Var5.Text;
ADOTable2.FieldByName('Вариант 6').AsString:=V_Var6.Text;

V_Var2.Clear;
V_Var3.Clear;
V_Var4.Clear;

V_Var6.Clear;
